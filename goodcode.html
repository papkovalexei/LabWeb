                #include <iostream>
                 
                template <typename... Arguments>
                class HW;
                 
                 
                template <class T>
                class HW<T>
                {
                public:
                    HW(T t): ch(t) {}
                    ~HW() { std::cout << ch;}
                private:
                    T ch;
                };
                 
                template <typename T, typename... ARGS>
                class HW<T, ARGS...> : public HW<ARGS...>
                {
                public:
                    HW(T t, ARGS... args):HW<ARGS...>(args...), ch(t){}
                    ~HW() { std::cout << ch;}
                private:
                    T ch;
                };
                 
                template <typename... ARGS>
                void Print(ARGS... args)
                {
                    HW<ARGS...> hw(args...);
                    (void)hw;
                }
